 module gpio_simple (
    input         clk,
    input         rst_n,
    input         wen,
    input  [31:0] wdata,
    output [31:0] rdata,
    output [31:0] gpio_out
);

    reg [31:0] gpio_reg;

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            gpio_reg <= 32'h0;
        else if (wen)
            gpio_reg <= wdata;
    end

    assign rdata    = gpio_reg;
    assign gpio_out = gpio_reg;

endmodule
